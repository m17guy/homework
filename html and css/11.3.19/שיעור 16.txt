הגדרת צמתים-

בעזרת הDOM, אנחנו יכולים ליצור, להכניס, למקם צמתים

var nodeToAdd = document.createElement("p"); // יוצרת פסקה חדשה, מכניסה את האובייקט לתוך המשתנה
nodeToAdd.setAttribute("class", "regular"); // <p class="regular"></p>

var newTxt = document.createTextNode("Hello!"); // צומת חדשה, שמכילה טקסט Hello!

nodeToAdd.appendChild(newTxt); // <p class="regular">Hello!</p>

var parentDiv = document.getElementByid("myDiv");
parentDiv.appendChild(nodeToAdd);
